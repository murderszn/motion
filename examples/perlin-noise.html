2D Classic Perlin Noise

Stefan Gustavson Textureless GPU Implementation
// Fade curve
vec2 fade(vec2 t) {
  return t*t*t*(t*(t*6.0-15.0)+10.0);
}

// Hash coordinate
vec4 permute(vec4 x) {
  return mod(((x*34.0)+1.0)*x, 289.0);
}

// 2D Classic Perlin Noise
float cnoise(vec2 P) {
  vec4 Pi = floor(P.xyxy) + vec4(0.,0.,1.,1.);
  vec4 Pf = fract(P.xyxy) - vec4(0.,0.,1.,1.);
  Pi = mod(Pi, 289.0);
  vec4 ix = Pi.xzxz; vec4 iy = Pi.yyww;
  vec4 fx = Pf.xzxz; vec4 fy = Pf.yyww;
  vec4 i = permute(permute(ix) + iy);
  vec4 gx = fract(i * (1.0 / 41.0)) * 2. - 1.;
  vec4 gy = abs(gx) - 0.5;
  vec4 tx = floor(gx + 0.5);
  gx = gx - tx;
  vec2 g00 = vec2(gx.x,gy.x);
  vec2 g10 = vec2(gx.y,gy.y);
  vec2 g01 = vec2(gx.z,gy.z);
  vec2 g11 = vec2(gx.w,gy.w);
  vec4 norm = 1.7928 - 0.8537 * 
    vec4(dot(g00,g00), dot(g01,g01),
         dot(g10,g10), dot(g11,g11));
  g00 *= norm.x; g01 *= norm.y;
  g10 *= norm.z; g11 *= norm.w;
  float n00 = dot(g00, vec2(fx.x,fy.x));
  float n10 = dot(g10, vec2(fx.y,fy.y));
  float n01 = dot(g01, vec2(fx.z,fy.z));
  float n11 = dot(g11, vec2(fx.w,fy.w));
  vec2 fade_xy = fade(Pf.xy);
  vec2 n_x = mix(vec2(n00,n01),
                 vec2(n10,n11), fade_xy.x);
  return 2.3 * mix(n_x.x, n_x.y, fade_xy.y);
}

This example implements Stefan Gustavson's textureless 2D Classic Perlin Noise.

Why Textureless?

Traditional noise implementations read precomputed numbers from 2D/3D textures. Textureless noise computes gradients procedurally on-the-fly, reducing texture binding swaps and memory bandwidth.

Quintic Hermite Interpolation

This implementation utilizes the quintic fade curve:
s(t) = 6t⁵ - 15t⁴ + 10t³
This mathematically guarantees C² continuity, preventing lighting and derivative banding artifacts.
